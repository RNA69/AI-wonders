<script>

const data = [

{ id: "SDG 1: No Poverty", value: 1 },

{ id: "SDG 2: Zero Hunger", value: 2 },

{ id: "SDG 3: Good Health and Well-being", value: 3 },

{ id: "SDG 4: Quality Education", value: 4 },

{ id: "SDG 5: Gender Equality", value: 5 },

{ id: "SDG 9: Industry, Innovation, and Infrastructure", value: 9 },

{ id: "SDG 10: Reduced Inequalities", value: 10 },

{ id: "SDG 11: Sustainable Cities and Communities", value: 11 },

{ id: "SDG 16: Peace, Justice, and Strong Institutions", value: 16 },

{ id: "SDG 17: Partnerships for the Goals", value: 17 }

];

 

const width = 800;

const height = 600;

 

const svg = d3.select("#bubble-map")

.append("svg")

.attr("width", width)

.attr("height", height);

 

const bubble = d3.pack()

.size([width, height])

.padding(10);

 

const root = d3.hierarchy({ children: data })

.sum(d => d.value);

 

bubble(root);

 

const nodes = svg.selectAll(".node")

.data(root.children)

.enter()

.append("g")

.attr("class", "node")

.attr("transform", d => `translate(${d.x},${d.y})`);

 

nodes.append("circle")

.attr("class", "bubble")

.attr("r", d => d.r);

 

nodes.append("text")

.attr("class", "label")

.attr("dy", ".3em")

.text(d => d.data.id);

</script>
